50template<typename Scalar>
51struct scalar_normal_dist_op
52{
53 static osuCrypto::PRNG rng; // The uniform pseudo-random algorithm
54 mutable std::normal_distribution<double> norm; // gaussian combinator
55
56 EIGEN_EMPTY_STRUCT_CTOR(scalar_normal_dist_op)
57
58 template<typename Index>
59 inline const Scalar operator() (Index, Index = 0) const { return (Scalar) norm(rng); }
60 inline void seed(const osuCrypto::block &s) { rng.SetSeed(s); }
61};

76template<typename Scalar>
77class EigenMultivariateNormal
78{
79 Matrix<Scalar,Dynamic,Dynamic> _covar;
80 Matrix<Scalar,Dynamic,Dynamic> _transform;
81 Matrix< Scalar, Dynamic, 1> _mean;
82 internal::scalar_normal_dist_op<Scalar> randN; // Gaussian functor
83 bool _use_cholesky;
84 SelfAdjointEigenSolver<Matrix<Scalar,Dynamic,Dynamic> > _eigenSolver; // drawback: this creates a useless eigenSolver when using Cholesky decomposition, but it yields access to eigenvalues and vectors
85
86 public:
87 EigenMultivariateNormal(const Matrix<Scalar,Dynamic,1>& mean,const Matrix<Scalar,Dynamic,Dynamic>& covar,
88 const bool use_cholesky=false,const osuCrypto::block &seed=osuCrypto::sysRandomSeed())
89 :_use_cholesky(use_cholesky)
90 {
91 randN.seed(seed);
92 setMean(mean);
93 setCovar(covar);
94 }
95
96 void setMean(const Matrix<Scalar,Dynamic,1>& mean) { _mean = mean; }
97 void setCovar(const Matrix<Scalar,Dynamic,Dynamic>& covar)
98 {
99 _covar = covar;
100
101 // Assuming that we'll be using this repeatedly,
102 // compute the transformation matrix that will
103 // be applied to unit-variance independent normals
104
105 if (_use_cholesky)
106 {
107 Eigen::LLT<Eigen::Matrix<Scalar,Dynamic,Dynamic> > cholSolver(_covar);
108 // We can only use the cholesky decomposition if
109 // the covariance matrix is symmetric, pos-definite.
110 // But a covariance matrix might be pos-semi-definite.
111 // In that case, we'll go to an EigenSolver
112 if (cholSolver.info()==Eigen::Success)
113 {
114 // Use cholesky solver
115 _transform = cholSolver.matrixL();
116 }
117 else
118 {
119 throw std::runtime_error("Failed computing the Cholesky decomposition. Use solver instead");
120 }
121 }
122 else
123 {
124 _eigenSolver = SelfAdjointEigenSolver<Matrix<Scalar,Dynamic,Dynamic> >(_covar);
125 _transform = _eigenSolver.eigenvectors()*_eigenSolver.eigenvalues().cwiseMax(0).cwiseSqrt().asDiagonal();
126 }
127 }
128
131 Matrix<Scalar,Dynamic,-1> samples(int nn)
132 {
133 Matrix<Scalar,Dynamic,Dynamic> foo = Matrix<Scalar, Dynamic, -1>::NullaryExpr(_covar.rows(), nn, randN);
134 return (_transform * foo).colwise() + _mean;
135 }
136}; // end class EigenMultivariateNormal
137} // end namespace Eigen
